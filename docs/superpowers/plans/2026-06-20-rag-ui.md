# RAG Search UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a browser UI to the existing RAG pipeline so users can query their knowledge base at `http://localhost:8765`.

**Architecture:** Three new FastAPI endpoints are added to `mcp_server.py` (`GET /`, `GET /indexes`, `POST /query`). A new `ui.html` file is served at `/`. The MCP SSE endpoints are untouched. The UI fetches indexes on load, lets the user toggle chips to filter indexes, and POSTs questions to `/query` which returns a full LLM-generated answer.

**Tech Stack:** FastAPI (already present), Python 3, vanilla JS + HTML/CSS (no frontend build step)

> Note: `list_indexed` is already imported from `query` — remove the duplicate if present.

Add the Pydantic model and endpoint (after `@app.get("/indexes")`):

```python
class QueryRequest(BaseModel):
    question: str
    indexes: Optional[list[str]] = None


@app.post("/query")
async def query_endpoint(req: QueryRequest):
    if not check_ollama_running():
        return {"error": "Ollama is not running. Start it with: ollama serve"}

    index_names = req.indexes if req.indexes else list_indexed()
    if not index_names:
        return {"error": "No indexes found. Run `python ingest.py --index <name>` first."}

    try:
        engine = build_query_engine(index_names)
        response = engine.query(req.question)
        answer = response.response or "Could not generate a summary. See sources below."
        sources = format_sources(response.source_nodes)
        return {"answer": answer, "sources": sources}
    except Exception as e:
        return {"error": str(e)}
```

  <script>
    let selected = new Set();

    async function loadIndexes() {
      const chipsEl = document.getElementById('chips');
      try {
        const res = await fetch('/indexes');
        const indexes = await res.json();
        if (indexes.length === 0) {
          chipsEl.innerHTML =
            '<span class="no-indexes">No indexes found — run <code>python ingest.py --index &lt;name&gt;</code> first.</span>';
          return;
        }
        chipsEl.innerHTML = indexes
          .map(name => `<span class="chip" data-name="${name}" onclick="toggleChip(this)">📁 ${name}</span>`)
          .join('');
      } catch {
        chipsEl.innerHTML =
          '<span class="no-indexes" style="color:#f85149">Could not reach server at /indexes.</span>';
      }
    }

    function toggleChip(el) {
      const name = el.dataset.name;
      if (selected.has(name)) {
        selected.delete(name);
        el.classList.remove('selected');
      } else {
        selected.add(name);
        el.classList.add('selected');
      }
      const statusEl = document.getElementById('status');
      if (selected.size === 0) {
        statusEl.textContent = 'Searching: all indexes';
        statusEl.className = 'status';
      } else {
        statusEl.textContent = 'Searching: ' + [...selected].join(', ');
        statusEl.className = 'status filtered';
      }
    }

    const questionEl = document.getElementById('question');
    const askBtn = document.getElementById('ask-btn');

    questionEl.addEventListener('input', () => {
      askBtn.disabled = questionEl.value.trim() === '';
    });
    questionEl.addEventListener('keydown', e => {
      if (e.key === 'Enter' && !askBtn.disabled) ask();
    });
    askBtn.addEventListener('click', ask);

    async function ask() {
      const question = questionEl.value.trim();
      if (!question) return;

      askBtn.disabled = true;
      askBtn.innerHTML = '<span class="spinner"></span>Asking…';

      const resultEl = document.getElementById('result');
      const answerEl = document.getElementById('answer');
      const sourcesEl = document.getElementById('sources');
      const sourcesSectionEl = document.getElementById('sources-section');
      resultEl.classList.remove('visible');

      try {
        const res = await fetch('/query', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            question,
            indexes: selected.size > 0 ? [...selected] : null,
          }),
        });
        const data = await res.json();

        if (data.error) {
          answerEl.innerHTML = `<span class="error-text">${data.error}</span>`;
          sourcesSectionEl.style.display = 'none';
        } else {
          answerEl.textContent = data.answer;
          sourcesEl.textContent = data.sources;
          sourcesSectionEl.style.display = 'block';
        }
      } catch (e) {
        answerEl.innerHTML = `<span class="error-text">Request failed: ${e.message}</span>`;
        sourcesSectionEl.style.display = 'none';
      }

      resultEl.classList.add('visible');
      askBtn.disabled = false;
      askBtn.textContent = 'Ask';
    }

    loadIndexes();
  </script>
</body>
</html>
